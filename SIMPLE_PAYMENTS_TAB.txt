const renderPaymentsTab = () => {
  console.log('Payment Tab Data:', { paymentLoading, paymentData, unpaidCount });
  
  return (
    <div className={styles.paymentsContainer}>
      <h2 className={styles.tabTitle}>Monthly Payments</h2>

      {paymentLoading && <div>Loading...</div>}

      {!paymentLoading && paymentData.length === 0 && (
        <div className={styles.emptyState}>
          <FiDollarSign className={styles.emptyIcon} />
          <p>No payment information available</p>
        </div>
      )}

      {!paymentLoading && paymentData.length > 0 && (
        <div>
          <div className={styles.paymentSummaryCard}>
            <h3>Payment Summary</h3>
            <p>Total Wards: {paymentData.length}</p>
            <p>Unpaid Invoices: {unpaidCount}</p>
          </div>
          
          {paymentData.map((wardPayment, index) => (
            <div key={index} className={styles.paymentCard}>
              <h4>{wardPayment.ward.studentName}</h4>
              <p>Class: {wardPayment.ward.class}</p>
              <p>Total Invoices: {wardPayment.summary.totalInvoices}</p>
              <p>Paid: {wardPayment.summary.paidInvoices}</p>
              <p>Unpaid: {wardPayment.summary.unpaidInvoices}</p>
              <p>Balance: {wardPayment.summary.totalBalance} ETB</p>
            </div>
          ))}
        </div>
      )}
    </div>
  );
};
